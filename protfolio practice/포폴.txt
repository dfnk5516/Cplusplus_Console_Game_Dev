void mainGame::SetColor(int color, int bgcolor)
{
	bgcolor &= 0xf;
	color &= 0xf;

	SetConsoleTextAttribute(GetStdHandle(STD_OUTPUT_HANDLE), color | (bgcolor << 4));
}

/////////////////////////////////////////////

void mainGame::BootScreen()
{
	SetColor(0, 0);
	BootScreen2();
	Sleep(500);

	SetColor(0, 15);
	BootScreen2();
	Sleep(1500);

	SetColor(3, 7);
	BootScreen2();
	Sleep(1000);

	SetColor(3, 7);
	BootScreen1();
	Sleep(1100);

	SetColor(3, 8);
	BootScreen1();
	Sleep(1200);

	SetColor(9, 8);
	BootScreen1();
	
	while (true)
	{
		_key = _getch();

		if (_key == (char)ENTER | _key == (char)SPACE)
		{
			SetColor(3, 8);
			BootScreen1();
			Sleep(1300);

			SetColor(3, 7);
			BootScreen1();
			Sleep(1200);

			SetColor(3, 7);
			BootScreen2();
			Sleep(1100);

			SetColor(0, 15);
			BootScreen2();
			Sleep(950);

			SetColor(0, 0);
			BootScreen2();
			Sleep(1500);

			break;
		}
	}
}

void mainGame::StartChoice()
{
	while(true)
	{
		_key = _getch()

		if (key == (char)LEFT || key == 'a' || key == 'A')
		{
			_start = 1;
		}
		else if (key == (char)RIGHT || key == 'd' || key == 'D')
		{
			_start = 0;
		}
		else if (key == (char)ENTER || key == (char)SPACE)
		{
			break;
		}
	}
}

void mainGame::Music()
{
	while (b_gameMusic == true)
	{
		while (b_startMusic == true)
		{
			auto start = chrono::high_resolution_clock::now();
			PlaySound(TEXT("Ю.wav"), NULL, SND_ASYNC);

			for (int i = 0; i < 30; i++)
			{
				if (b_startMusic == false)
				{
					PlaySound(NULL,NULL,0);
					break;
				}
				this_thread::sleep_until(start + (i + 1)*std::chrono::milliseconds(985));
			}

		}
	}
}
